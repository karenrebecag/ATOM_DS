# ATOM Design System — Component Architecture Guide

**Versión:** 1.0.0
**Última actualización:** 2026-03-29
**Stack:** pnpm + Turborepo + Style Dictionary v4 + Astro + GSAP 3.12

---

## 📐 Filosofía Central

ATOM usa una arquitectura de **separación total de capas** que permite:

- ✅ Componentes framework-agnostic (Astro, React, Vue, Vanilla)
- ✅ Animaciones reutilizables sin acoplamiento al componente
- ✅ Diseño basado en tokens (single source of truth)
- ✅ CSS puro sin runtime JS (solo animaciones opcionales)

---

## 🏗️ Las 4 Capas del Sistema

```
┌─────────────────────────────────────────────────────────┐
│ 1. TOKENS (Design Decisions)                           │
│    JSON (W3C DTCG) → CSS Custom Properties              │
│    @atomchat/tokens                                     │
└─────────────────────────────────────────────────────────┘
                          ↓
┌─────────────────────────────────────────────────────────┐
│ 2. CSS (Visual Implementation)                          │
│    Pure CSS with BEM → Consumes var(--tokens)           │
│    @atomchat/css                                        │
└─────────────────────────────────────────────────────────┘
                          ↓
┌─────────────────────────────────────────────────────────┐
│ 3. COMPONENTS (HTML Structure)                          │
│    Astro/React/Vue → Props + HTML + data-attributes     │
│    @atomchat/components-astro                           │
└─────────────────────────────────────────────────────────┘
                          ↓
┌─────────────────────────────────────────────────────────┐
│ 4. ANIMATIONS (Optional Behavior)                       │
│    GSAP + data-attribute contract → Runtime behavior    │
│    @atomchat/animations                                 │
└─────────────────────────────────────────────────────────┘
```

### Reglas de Oro

1. **Tokens** → NO saben de CSS, componentes, ni animaciones
2. **CSS** → Consume tokens, NO sabe de componentes ni animaciones
3. **Components** → Consumen tokens + CSS, NO importan animaciones
4. **Animations** → Descubren componentes via `data-*`, NO importan componentes

### PASO 3: Component (HTML Structure)

**Ubicación:** `packages/components-astro/src/atoms/Badge.astro`

```astro
---
/**
 * Badge (Notification Badge)
 *
 * A notification indicator with smart overflow logic.
 * Displays notification counts with context-aware capping rules.
 *
 * @param {number} count - The raw notification count
 * @param {string} context - Context for overflow rules: "default" | "inbox"
 * @param {string} type - Visual type: "neutral" | "inbox" | "info"
 * @param {string} state - Visual state: "default" | "focused" | "subtle"
 * @param {string} class - Additional CSS classes
 *
 * Overflow Rules:
 * - Default context: Shows exact count 1-99, then "99+"
 * - Inbox context: Shows exact count 1-49, then "+50"
 * - Zero count: Hidden (returns null)
 *
 * Usage:
 * <Badge count={45} context="inbox" type="inbox" state="focused" />
 * <Badge count={150} context="default" type="info" />
 */

interface Props {
  count: number;
  context?: "default" | "inbox";
  type?: "neutral" | "inbox" | "info";
  state?: "default" | "focused" | "subtle";
  class?: string;
  [key: string]: any;
}

const {
  count = 0,
  context = "default",
  type = "neutral",
  state = "default",
  class: className = "",
  ...rest
} = Astro.props;

// ── Badge Display Logic (Overflow Rules) ──
function getBadgeDisplay(count: number, context: string): string | null {
  // Rule 4: Zero — hide
  if (count <= 0) return null;

  if (context === "inbox") {
    // Inbox context rules
    if (count >= 50) return "+50"; // Rule 2: Inbox cap
    return count.toString();        // Rule 1: 1-49 exact
  }

  // Default context rules
  if (count > 99) return "99+";     // Rule 2: Default cap
  return count.toString();          // Rule 1: 1-99 exact
}

const displayValue = getBadgeDisplay(count, context);

// If count is zero or invalid, don't render anything
if (!displayValue) {
  return null;
}

const ariaLabel = `${count} ${count === 1 ? 'notification' : 'notifications'}`;
---

<span
  class:list={[
    "badge",
    `badge--${type}`,
    `badge--${state}`,
    className,
  ]}
  data-badge
  data-count={count}
  data-context={context}
  data-state={state}
  role="status"
  aria-label={ariaLabel}
  {...rest}
>
  <span class="badge__text" data-badge-text>{displayValue}</span>
</span>
```

### PASO 4: Animations (Optional Behavior)

**Ubicación:** `packages/animations/src/badge.ts`

```typescript
/**
 * @module badge
 * Reusable GSAP animations for badge components.
 */

import { gsap, DURATION } from "./config";
import { prefersReducedMotion, isMotionExempt } from "./motion";
import type { AnimationConfig, AnimationOptions, CleanupFn } from "./types";
import { NOOP } from "./types";
import { initNumberOdometer } from "./odometer";

/**
 * Hide badge with scale-out animation.
 */
export function hideBadge(
  element: HTMLElement,
  options: AnimationOptions = {},
): CleanupFn {
  if (isMotionExempt(element)) return NOOP;

  if (prefersReducedMotion()) {
    gsap.set(element, { scale: 0, opacity: 0 });
    return NOOP;
  }

  const tween = gsap.to(element, {
    scale: 0,
    opacity: 0,
    duration: options.duration ?? DURATION.half,
    ease: options.ease ?? "power2.in",
    delay: options.delay,
  });

  return () => tween.kill();
}

/**
 * Show badge with scale-in animation.
 */
export function showBadge(
  element: HTMLElement,
  options: AnimationOptions = {},
): CleanupFn {
  if (isMotionExempt(element)) return NOOP;

  if (prefersReducedMotion()) {
    gsap.set(element, { scale: 1, opacity: 1 });
    return NOOP;
  }

  const tween = gsap.from(element, {
    scale: 0,
    opacity: 0,
    duration: options.duration ?? 0.4,
    ease: options.ease ?? "back.out(1.7)",
    delay: options.delay,
  });

  return () => tween.kill();
}

/**
 * Entry animation for badge (on initial mount).
 */
export function animateBadgeEntry(
  element: HTMLElement,
  options: AnimationOptions = {},
): CleanupFn {
  return showBadge(element, options);
}

/**
 * Initialize badge lifecycle management.
 *
 * Sets up MutationObserver for all `[data-badge]` elements to:
 * - Watch `data-count` attribute changes
 * - Animate count updates with odometer effect
 * - Show/hide badge when count changes to/from 0
 * - Trigger entry animation on mount if count > 0
 *
 * @param config - Optional scope config
 * @returns Cleanup function that disconnects all observers
 */
export function initBadge(config: AnimationConfig = {}): CleanupFn {
  const scope = config.scope || document;
  const badges = scope.querySelectorAll<HTMLElement>("[data-badge]");

  if (!badges.length) return NOOP;

  // Initialize odometer system for count updates
  const updateOdometer = initNumberOdometer();

  // Track observers for cleanup
  const observers: MutationObserver[] = [];
  const cleanups: CleanupFn[] = [];

  badges.forEach((badge) => {
    const badgeText = badge.querySelector<HTMLElement>("[data-badge-text]");
    if (!badgeText) return; // ← Skip if no data-badge-text

    let currentCount = parseInt(badge.dataset.count || "0", 10);
    const context = badge.dataset.context || "default";

    // Helper to calculate display value
    function getDisplayValue(count: number, context: string): string {
      if (count <= 0) return "0";
      if (context === "inbox") {
        if (count >= 50) return "+50";
        return count.toString();
      }
      if (count > 99) return "99+";
      return count.toString();
    }

    // Observe attribute changes
    const observer = new MutationObserver((mutations) => {
      mutations.forEach((mutation) => {
        if (mutation.type === "attributes" && mutation.attributeName === "data-count") {
          const newCount = parseInt(badge.dataset.count || "0", 10);

          if (newCount !== currentCount) {
            const newDisplay = getDisplayValue(newCount, context);

            // Animate count change with odometer
            updateOdometer(badgeText, newDisplay, { duration: 0.4 });

            // Update aria-label
            const ariaLabel = `${newCount} ${newCount === 1 ? 'notification' : 'notifications'}`;
            badge.setAttribute("aria-label", ariaLabel);

            // Hide badge if count becomes 0
            if (newCount === 0 && currentCount > 0) {
              cleanups.push(hideBadge(badge));
            }
            // Show badge if count was 0 and now > 0
            else if (newCount > 0 && currentCount === 0) {
              cleanups.push(showBadge(badge));
            }

            currentCount = newCount;
          }
        }
      });
    });

    observer.observe(badge, {
      attributes: true,
      attributeFilter: ["data-count"],
    });

    observers.push(observer);

    // Entry animation for badges visible on mount (count > 0)
    if (currentCount > 0) {
      cleanups.push(animateBadgeEntry(badge));
    }
  });

  // Return cleanup that disconnects all observers
  return () => {
    observers.forEach((observer) => observer.disconnect());
    cleanups.forEach((cleanup) => cleanup());
  };
}
```

## 🔌 Data-Attribute Contract

### ¿Por qué data-attributes?

Los `data-*` attributes son el **contrato** entre componentes y animaciones:

- ✅ Framework-agnostic (funciona en Astro, React, Vue, Vanilla)
- ✅ No requiere imports entre capas
- ✅ Testeable con `querySelector`
- ✅ HTML válido y semántico

### Naming Convention

```
data-{component}              ← Root element (required)
data-{component}-{element}    ← Animatable child elements
```

**Ejemplos:**

| Componente | Root | Children |
|------------|------|----------|
| Badge | `data-badge` | `data-badge-text` |
| Button | `data-button` | `data-button-icon`, `data-button-label` |
| Chip | `data-chip` | `data-chip-text`, `data-chip-close` |
| Toggle | `data-toggle` | `data-toggle-track`, `data-toggle-thumb` |

### Atributos Estándar

| Atributo | Propósito | Ejemplo |
|----------|-----------|---------|
| `data-{component}` | Marca el componente para descubrimiento | `data-badge` |
| `data-count` | Datos reactivos (watched por MutationObserver) | `data-count="42"` |
| `data-context` | Variante de comportamiento | `data-context="inbox"` |
| `data-state` | Estado visual (opcional, solo CSS) | `data-state="focused"` |
| `data-motion-exempt` | Skip animación (a11y) | `data-motion-exempt` |

import { useEffect, useRef, useState } from "react";
import { initBadge } from "@atomchat/animations";

function BadgeDemo() {
  const badgeRef = useRef<HTMLSpanElement>(null);
  const [count, setCount] = useState(5);

  // Calculate display value (same logic as Astro component)
  function getDisplayValue(count: number, context: string) {
    if (count <= 0) return null;
    if (context === "inbox") {
      if (count >= 50) return "+50";
      return count.toString();
    }
    if (count > 99) return "99+";
    return count.toString();
  }

  const displayValue = getDisplayValue(count, "default");

  // Sync count to data-attribute (MutationObserver will pick it up)
  useEffect(() => {
    if (badgeRef.current) {
      badgeRef.current.setAttribute("data-count", count.toString());
    }
  }, [count]);

  // Initialize animations
  useEffect(() => {
    const cleanup = initBadge();
    return cleanup;
  }, []);

  if (!displayValue) return null;

  return (
    <div>
      <span
        ref={badgeRef}
        className="badge badge--neutral badge--default"
        data-badge
        data-count={count}
        data-context="default"
        data-state="default"
        role="status"
        aria-label={`${count} notifications`}
      >
        <span className="badge__text" data-badge-text>
          {displayValue}
        </span>
      </span>

      <button onClick={() => setCount(c => c + 1)}>+</button>
      <button onClick={() => setCount(c => Math.max(0, c - 1))}>−</button>
    </div>
  );
}

### 4. Animations

```bash
[ ] Crear packages/animations/src/{nombre}.ts
[ ] Exportar initNombre(): CleanupFn

[ ] Retornar cleanup function que desconecta observers y mata tweens

[ ] Build: cd packages/animations && pnpm build
[ ] Test en proyecto consumidor
```

**Template:**
```typescript
/**
 * @module component-name
 * Brief description of animations
 */

import { gsap } from "./config";
import { prefersReducedMotion } from "./motion";
import type { AnimationConfig, CleanupFn } from "./types";
import { NOOP } from "./types";

/**
 * Initialize component-name animations.
 */
export function initComponentName(config: AnimationConfig = {}): CleanupFn {
  if (prefersReducedMotion()) return NOOP;

  const scope = config.scope || document;
  const components = scope.querySelectorAll<HTMLElement>("[data-component-name]");

  if (!components.length) return NOOP;

  const tweens: gsap.core.Tween[] = [];

  components.forEach((el) => {
    const text = el.querySelector<HTMLElement>("[data-component-name-text]");
    if (!text) return;

    // Your animation logic here
    const tween = gsap.from(text, {
      opacity: 0,
      y: 20,
      duration: 0.6,
      ease: "power2.out",
    });

    tweens.push(tween);
  });

  // Return cleanup
  return () => {
    tweens.forEach((t) => t.kill());
  };
}
```

## 🚫 Anti-Patterns y Errores Comunes

### ❌ Anti-Pattern 1: Hardcoded Values en CSS

```css
/* ❌ MAL */
.badge {
  padding: 0 2px;
  background: #f4f4f5;
  font-size: 10px;
}

/* ✅ BIEN */
.badge {
  padding: var(--badge-padding-y) var(--badge-padding-x);
  background: var(--badge-bg-neutral-default);
  font-size: var(--badge-font-size);
}
```

---

### ❌ Anti-Pattern 2: Componente con <style> inline

```astro
<!-- ❌ MAL -->
<span class="badge">...</span>

<style>
  .badge {
    background: var(--badge-bg);
  }
</style>

<!-- ✅ BIEN -->
<span class="badge">...</span>
<!-- Estilos vienen de @atomchat/css -->
```

---

### ❌ Anti-Pattern 3: Componente importa animaciones

```astro
<!-- ❌ MAL -->
<span class="badge" data-badge>...</span>

<script>
  import { initBadge } from "@atomchat/animations";
  initBadge();
</script>

<!-- ✅ BIEN -->
<span class="badge" data-badge>...</span>
<!-- Animaciones se inicializan en Layout con initAllAnimations() -->
```

---

### ❌ Anti-Pattern 4: Olvidar data-attributes

```astro
<!-- ❌ MAL (animaciones no funcionarán) -->
<span class="badge">
  <span class="badge__text">5</span>
</span>

<!-- ✅ BIEN -->
<span class="badge" data-badge data-count="5">
  <span class="badge__text" data-badge-text>5</span>
</span>
```

---

### ❌ Anti-Pattern 5: Duplicar archivos locales en vez de importar del monorepo

```
❌ MAL:
my-project/
  src/
    animations/     ← Copia local de @atomchat/animations
      badge.ts
      odometer.ts
      ...

✅ BIEN:
my-project/
  package.json    ← "@atomchat/animations": "^1.0.0"
  src/
    layouts/
      Layout.astro  ← import { initAllAnimations } from "@atomchat/animations"
```

---

### ❌ Anti-Pattern 6: Actualizar data-count solo en state, no en DOM

```tsx
// ❌ MAL (MutationObserver no detectará cambios)
function Badge() {
  const [count, setCount] = useState(5);

  return (
    <span data-badge data-count="5">  {/* ← Hardcoded */}
      <span data-badge-text>{count}</span>
    </span>
  );
}

// ✅ BIEN (sync state → DOM attribute)
function Badge() {
  const badgeRef = useRef<HTMLSpanElement>(null);
  const [count, setCount] = useState(5);

  useEffect(() => {
    if (badgeRef.current) {
      badgeRef.current.setAttribute("data-count", count.toString());
    }
  }, [count]);

  return (
    <span ref={badgeRef} data-badge data-count={count}>
      <span data-badge-text>{count}</span>
    </span>
  );
}
```
